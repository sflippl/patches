class CompositionalBinaryData(torch.utils.data.Dataset):
    """Provides video data following the Pytorch Dataset framework.
    """
    def __init__(self, cbm, contrastive_samples=1, use_seed=True,
                 predictions=range(1, 2)):
        self.cbm = cbm
        self.contrastive_samples = contrastive_samples
        self.predictions = predictions
        if use_seed:
            self.random_state = cbm.random_state
        else:
            self.random_state = np.random.RandomState()

    def sample(self, samples=0):
        """Continue sampling the compositional binary data.
        """
        self.cbm.sample(samples)

    def __getitem__(self, idx):
        if torch.is_tensor(idx):
            idx = idx.tolist()
        arrs = []
        predictions = []
        contrastive_samples = []
        for __ in range(self.contrastive_samples):
            contrastive_samples.append([])
        contrastive_idxs = torch.randint(0, len(self), (self.contrastive_samples, ))
        for atom in self.cbm.atoms:
            arr = np.zeros(shape=(1, atom.width))
            arr[0, atom.locations[idx][0]] = atom.values[idx]
            arrs.append(arr)
            prediction = np.zeros(shape=(len(self.predictions), atom.width))
            for i, pred_idx in enumerate(self.predictions):
                prediction[i, atom.locations[pred_idx][0]] = atom.values[pred_idx]
            predictions.append(prediction)
            for c_idx, contrastive_sample in zip(contrastive_idxs,
                                                 contrastive_samples):
                arr = np.zeros(shape=(1, atom.width))
                arr[0, atom.locations[c_idx][0]] = atom.values[c_idx]
                contrastive_sample.append(arr)
        arr = np.concatenate(arrs, axis=1)
        contrast = np.concatenate([
            np.concatenate(c_arrs, axis=1) for c_arrs in contrastive_samples
        ], axis=0)
        predicted = np.concatenate(predictions, axis=1)
        return {'sample': torch.tensor(arr.astype('float32'), requires_grad=True),
                'predicted': torch.tensor(predicted.astype('float32'), requires_grad=True),
                'contrast': torch.tensor(contrast.astype('float32'), requires_grad=True)}

    def __len__(self):
        return len(self.cbm.atoms[0].locations) - max(self.predictions)

class ContrastivePredictiveNet(nn.Module):
    def __init__(self, timesteps=10):
        super().__init__()
        self.encoder = nn.Linear(2, 1, bias=False)
        #self.predictor = nn.Linear(1, timesteps, bias=False)
        self.predictor = torch.tensor(-1., dtype=torch.float, requires_grad=True)

    def forward(self, x):
        encoded = {
            'sample': self.encoder(x['sample']),
            'contrast': self.encoder(x['contrast']),
            'predicted': self.encoder(x['predicted'])
        }
        encoded['prediction'] = self.predictor(encoded['sample'])
        return encoded

def criterion(x, y):
    return torch.data(x[:,0], y[:,0])

def crit(predictions):
    pred_loss = torch.exp(criterion(predictions['sample'],
                                    predictions['predicted']))
    contrast_loss = pred_loss + torch.sum(
        torch.exp(
            predictions['contrast']*predictions['sample']
        )
    )
    return -torch.log(pred_loss)+torch.log(contrast_loss)

for epoch in range(10):
    running_loss = 0.0
    for i, data in enumerate(cbd):
        inputs = data
        optimizer.zero_grad()
        predictions = cpn(inputs)
        loss = crit(predictions)
        loss.backward()
        optimizer.step()

        running_loss += loss.item()
        if i % 500 == 499:    # print every 2000 mini-batches
            print('[%d, %5d] loss: %.3f' %
                  (epoch + 1, i + 1, running_loss / 500))
            running_loss = 0.0
