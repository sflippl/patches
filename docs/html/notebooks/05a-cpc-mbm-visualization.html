import patches
import lettertask
import torch.optim as optim
from tqdm import tqdm
import numpy as np
import plotnine as gg
import lazytools_sflippl as lazytools
import torch.nn as nn
import torch

cbms = [
    lettertask.data.CompositionalBinaryModel(
        width=width, change_probability=change_probability,
        samples=10000, seed=2002
    ) for width, change_probability in zip(
        [[5, 5], [5, 5], [50, 50], [50, 50],
         [5, 10], [10, 50], [5, 5, 5], [10, 10, 10]],
        [[0.05, 0.5], [0.05, 0.2], [0.05, 0.5], [0.05, 0.2],
         [0.2, 0.05], [0.2, 0.1], [0.05, 0.1, 0.5], [0.05, 0.1, 0.5]]
    )
]

def regime(method, lr):
    def optimizer(params):
        return method(params, lr=lr)
    return optimizer
optimization_regimes = []
for method in [optim.SGD, optim.Adam, optim.Adadelta]:
    for lr in [1e-2, 1e-1, 1]:
        optimization_regimes.append(regime(method, lr))

loss_dfs = []
angle_dfs = []

with tqdm(total = len(optimization_regimes)*len(cbms)*4) as pbar:
    for idx_cbm, cbm in enumerate(cbms):
        for idx_opt, opt in enumerate(optimization_regimes):
            for method in ['barec', 'lapred1p', 'lapred2p', 'cpc']:
                angles = np.load(
                    '../experiments/04-data/arrays/cbm_{}_opt_{}_{}_angles.npy'.\
                    format(idx_cbm, idx_opt, method)
                )
                loss = np.load(
                    '../experiments/04-data/arrays/cbm_{}_opt_{}_{}_loss.npy'.\
                    format(idx_cbm, idx_opt, method)
                )
                loss_df = lazytools.array_to_dataframe(loss).\
                                    rename(columns={'dim0': 'time',
                                                    'array': 'loss'})
                loss_df['time'] *= 50
                loss_df['cbm_idx'] = np.array(idx_cbm)
                loss_df['opt_idx'] = np.array(idx_opt)
                loss_df['method'] = np.array(method)
                loss_df['cbm'] = cbm.title()
                loss_df['optimizer'] = np.array(str(opt([torch.tensor(1)])), dtype=str)
                loss_dfs.append(loss_df)
                angle_df = lazytools.array_to_dataframe(angles).\
                                      rename(columns={'dim0': 'latent_dimension',
                                                      'dim1': 'time',
                                                      'array': 'angle'})
                angle_df['time'] *= 10
                angle_df['cbm_idx'] = np.array(idx_cbm)
                angle_df['opt_idx'] = np.array(idx_opt)
                angle_df['latent_dimension'] = angle_df['latent_dimension'].\
                    astype(str)
                angle_df['method'] = np.array(method)
                angle_df['cbm'] = cbm.title()
                angle_df['optimizer'] = np.array(str(opt([torch.tensor(1)])), dtype=str)
                angle_dfs.append(angle_df)
                pbar.update(1)
