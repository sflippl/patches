class SupervisedCNN(nn.Module):
    def __init__(self, latent_features, input_features=None,
                 data=None, bias=True):
        super().__init__()
        if data:
            input_features = input_features or data.n_vars[0]*data.n_vars[1]
        elif input_features is None or timesteps is None:
            raise ValueError('You must either provide data or both input '
                             'features and timesteps.')
        self.latent_features = latent_features
        self.recognition = nn.Conv2d(1, 1, 2, padding=1)
        self.nonlinearity = nn.ReLU()
        self.encoder = nn.Linear(9, latent_features, bias=bias)

    def forward(self, x):
        latent = self.recognition(
            x['input'].reshape(
                x['input'].shape[0], 1, *x['input'].shape[1:]
            )
        )
        latent = self.nonlinearity(latent)
        latent = self.encoder(latent.flatten(1,-1))
        return latent

class SupervisedPredictiveCNN(nn.Module):
    def __init__(self, latent_features, input_features=None,
                 data=None, bias=True):
        super().__init__()
        if data:
            input_features = input_features or data.n_vars[0]*data.n_vars[1]
        elif input_features is None or timesteps is None:
            raise ValueError('You must either provide data or both input '
                             'features and timesteps.')
        self.latent_features = latent_features
        self.recognition = nn.Conv2d(1, 1, 2, padding=1)
        self.nonlinearity = nn.ReLU()
        self.encoder = nn.Linear(9, latent_features, bias=bias)
        self.predictor = nn.Linear(latent_features, 5*latent_features)

    def forward(self, x):
        latent = self.recognition(
            x['input'].reshape(
                x['input'].shape[0], 1, *x['input'].shape[1:]
            )
        )
        latent = self.nonlinearity(latent)
        latent = self.encoder(latent.flatten(1,-1))
        latent = self.predictor(latent).reshape(-1, 5, self.latent_features)
        return latent

class SupervisedCNN(nn.Module):
    def __init__(self, latent_features, input_features=None,
                 data=None, bias=True):
        super().__init__()
        if data:
            input_features = input_features or data.n_vars[0]*data.n_vars[1]
        elif input_features is None or timesteps is None:
            raise ValueError('You must either provide data or both input '
                             'features and timesteps.')
        self.latent_features = latent_features
        self.recognition = nn.Conv2d(1, 1, 2, padding=1)
        self.nonlinearity = nn.LeakyReLU()
        self.encoder = nn.Linear(36, latent_features, bias=bias)

    def forward(self, x):
        latent = self.recognition(
            x.reshape(-1, 1, *x.shape[3:])
        )
        latent = self.nonlinearity(latent)
        latent = self.encoder(latent.flatten(1,-1))
        return latent.reshape(*x.shape[:3], 1)

def reaction(parameters):
    rf = list(parameters.parameters())[0].detach().numpy()
    bias = list(parameters.parameters())[1].detach().numpy()
    shapes = [
        np.array([[1, 1], [0, 0]]),
        np.array([[0, 0], [1, 1]]),
        np.array([[1, 0], [1, 0]]),
        np.array([[0, 1], [0, 1]])
    ]
    reaction = []
    for arr in shapes:
        reaction.append(np.maximum((rf*arr).sum()+bias, 0))
    return np.array(reaction)
